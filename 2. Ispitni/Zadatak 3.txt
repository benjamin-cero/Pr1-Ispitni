#include <iostream>
using namespace std;

/*Napisati program koji omoguÄ‡ava unos pozitivnog broja n te pronalazi njegove parne cifre i formira novi broj u kojem su parne cifre zamijenjene cifrom '5'.

Dodatno program treba ispisati razliku unesenog i broja koji se dobije nakon zamjene.*/


int obrnutecifre(int n) {
	int novi = 0;
	int pozicija = 1;

	while (n > 0)
	{
		int cifra = n % 10;
		if (cifra % 2 == 0) {
			cifra = 5;
		}
		novi = novi + cifra * pozicija;
		pozicija *= 10;
		n /= 10;
	}
	return novi;
}



void main() {
	int n;

	do
	{
		cout << "Unesite n: ";
		cin >> n;

	} while (n <= 0);

	int novi = obrnutecifre(n);

	cout << "Novi broj: " << novi << endl;
	cout << "Razlika: " << n - novi << endl;

	system("pause");
}

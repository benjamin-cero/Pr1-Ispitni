int brojcifri(int broj) {
	int brcifri = 0;
	while (broj > 0)
	{
		broj /= 10;
		brcifri++;
	}
	return brcifri;
}

// 12345678
float srednja(int n) {
	int cifre = brojcifri(n);
	float srednja;
	int intsrednja;
	if (cifre % 2 == 0) {
		n = n / pow(10.0, (cifre / 2 - 1));
		srednja = n % 10;
		n /= 10;
		srednja += n % 10;
		return srednja / 2.0;
	}
	else {
		intsrednja = n / pow(10.0, cifre / 2);
		return intsrednja % 10;
	}
}
int pomnozi(int n) {
	float srednjacifra = srednja(n);
	float noviBroj = 0;
	int brojac = 1;
	while (noviBroj <= n) {
		noviBroj = srednjacifra * brojac;
		brojac++;
	}
	return brojac;
}

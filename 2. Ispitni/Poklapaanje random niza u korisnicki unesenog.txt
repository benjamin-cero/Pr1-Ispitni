/*Napišite program koji generiše niz od 5 nasumičnih brojeva (koristeći rand() funkciju). Program mora da prati pravila:
•

. Svi brojevi u nizu moraju biti različiti.

bool Postoji(int niz[], int vel, int broj)
{
    for (int i = 0; i < vel; i++)
        if (niz[i] == broj)
            return true;

    return false;
}

bool Jednaki(int a[], int b[], int vel)
{
    for (int i = 0; i < vel; i++)
        if (a[i] != b[i])
            return false;

    return true;
}

void main(){
    srand(time(0));

    const int N = 5;
    int trazeni[N];
    int korisnik[N];


    for (int i = 0; i < N; )
    {
        int broj = rand() % 5 + 1;

        if (!Postoji(trazeni, i, broj))
        {
            trazeni[i] = broj;
            i++;
        }
    }

    bool pobjeda = false;

    for (int pokusaj = 1; pokusaj <= 5; pokusaj++)
    {
        cout << "Pokusaj " << pokusaj << " - unesite 5 razlicitih brojeva (1-5):\n";

        for (int i = 0; i < N; )
        {
            cin >> korisnik[i];

            if (korisnik[i] < 1 || korisnik[i] > 5 || Postoji(korisnik, i, korisnik[i]))
            {
                cout << "Neispravan unos! Pokusajte ponovo: ";
            }
            else
            {
                i++;
            }
        }

        if (Jednaki(trazeni, korisnik, N))
        {
            pobjeda = true;
            cout << "\nCestitamo, pobijedili ste! Nizovi su jednaki." << endl;
            break;
        }
        else
        {
            cout << "Niste unijeli ispravno niz! Tacni su elementi na pozicijama: ";

            for (int i = 0; i < N; i++)
                if (trazeni[i] == korisnik[i])
                    cout << i + 1 << " ";

            cout << endl;
        }
    }

    if (!pobjeda)
        cout << "Niste uspjeli da rijesite zagonetku. Izgubili ste." << endl;

    cout << "Trazeni niz: ";
    for (int i = 0; i < N; i++) {
        cout << trazeni[i] << " ";
    }

    cout << "\nKorisnikov zadnji niz: ";
    for (int i = 0; i < N; i++) {
        cout << korisnik[i] << " ";
    }

    cout << endl;

    system("pause");
}
